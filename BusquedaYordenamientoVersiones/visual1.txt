    private void registroArregloActionPerformed(java.awt.event.ActionEvent evt) {                                                
        
        String c = txtRango.getText();
        String ma = txtMaximo.getText();
        String mi = txtMinimo.getText();
        
        int c1 = Integer.parseInt(c);
        int ma1 = Integer.parseInt(ma);
        int mil = Integer.parseInt(mi);
        
        a.setCantidad(c1);
        a.setMin(mil);
        a.setMax(ma1);
        a.creaArreglo();
        
        
        txtRango.setText("");
        txtMaximo.setText("");
        txtMinimo.setText("");
    }                                               

    private void bEscribeActionPerformed(java.awt.event.ActionEvent evt) {

    private void bBuscaActionPerformed(java.awt.event.ActionEvent evt) {                                       
        int s;
        
        String c = txtBusca.getText();
        int s1 = Integer.parseInt(c);
        txtBusca.setText("");
        s = a.busquedaSecuencial(s1);
        
        if(s!=-1){
            System.out.print("\n");
            System.out.print(a.m.get(s));
        }
        else{
            System.out.print("\nNo se encontro el numero :(");
        }
    }
